if playerprefs.GetInt("WindBoss") == 1 then
	conversation[[
	WindTablet TabletPic right : Thank you for freeing the Spirit of Wind!
	]]
else
	conversation[[
	WindTablet TabletPic right : He who tresspasses into his  domain must be careful not to anger him.
	WindTablet : His rage cuts like the sharpest of blades.
	]]

	say "Do you want to go to the wind elemental's dungeon?"

	function TeleportIn()
		luautils.LoadScene("WindDungeon")
	end

	menutitle "Wind Dungeon"

	local choice = choosetimer ({"Enter the Dungeon", "Leave"}, 5, 2)

	if choice == 1 then
		say "Good Luck!"
		TeleportIn()
	else
		say "That is probably wise..."
	end
end
